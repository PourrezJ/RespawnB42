description=[h1]Respawn - Build 42[/h1]
description=
description=[h2]Description[/h2]
description=This mod allows you to respawn after death while keeping a portion of your skills, traits, and profession!
description=
description=[b]Based on the original Respawn mod by StarkSilence:[/b]
description=[url=https://steamcommunity.com/sharedfiles/filedetails/?id=2687842971]Respawn (Build 41)[/url]
description=
description=This version has been fully adapted and updated for [b]Build 42[/b] with new features and improvements.
description=
description=[h2]Features[/h2]
description=[list]
description=[*] [b]Restore Your Progress:[/b] After dying, create a new character with the special "Respawn" profession to recover your previous stats
description=[*] [b]Customizable XP Recovery:[/b] Choose how much XP to restore (5% to 100%, or exact last level)
description=[*] [b]Trait Recovery:[/b] All your character traits are restored (excluding profession-specific traits)
description=[*] [b]Profession Recovery:[/b] Your original profession is fully restored
description=[*] [b]Recipe Knowledge:[/b] All learned recipes are carried over
description=[*] [b]Flexible Options:[/b] Configure Fitness and Strength to always restore at 100% regardless of XP settings
description=[*] [b]Multiplayer Support:[/b] Fully functional in multiplayer with server-controlled options
description=[/list]
description=
description=[h2]How to Use[/h2]
description=[olist]
description=[*] Play the game normally with any profession
description=[*] When you die, your character stats are automatically saved
description=[*] Create a new character and select the [b]"Respawn"[/b] profession
description=[*] Your previous profession, traits, XP, and recipes will be automatically restored!
description=[/olist]
description=
description=[h2]Configuration[/h2]
description=[b]Solo Mode:[/b]
description=Go to [b]Options > Mods > Respawn Mod[/b] to configure:
description=[list]
description=[*] [b]XP Restored:[/b] Choose the percentage of XP to restore (5%-100%) or "Last Level"
description=[*] [b]Exclude Fitness:[/b] Always restore 100% Fitness XP
description=[*] [b]Exclude Strength:[/b] Always restore 100% Strength XP
description=[/list]
description=
description=[b]Multiplayer/Server:[/b]
description=Server administrators can edit the configuration file:
description=[code]C:\Users\[User]\Zomboid\Lua\respawn-options.json[/code]
description=
description=Example configuration:
description=[code]{
description=    "XPRestored": 21,
description=    "ExcludeFitness": true,
description=    "ExcludeStrength": true
description=}[/code]
description=
description=[b]XPRestored values:[/b] 1 = 5%, 2 = 10%, ..., 20 = 100%, 21 = Last Level
description=
